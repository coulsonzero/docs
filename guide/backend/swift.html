func minMax(arr: [Int]) -> (min: Int, max: Int) {
    var curMin: Int = arr[0]
    var curMax: Int = arr[0]
    for v in arr[1..<arr.count] {
        if v < curMin {
            curMin = v
        } else if v > curMax {
            curMax = v
        }
    }
    return (curMin, curMax)
}

print(minMax(arr: [1, 3, 6, 2, -2]))    // (min: -2, max: 6)
