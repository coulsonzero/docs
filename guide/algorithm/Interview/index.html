const pathList = ["/a/b", "/a/2/c", "/d"]

function pathFormat(pathList) {
	const answer = {}
	var array = []
	for (let i = 0; i < pathList.length; i++) {
		// step1. '/a/2/c' => ['a', '2', 'c]
		let arr = pathList[i].split("/").filter((n) => n !== "")
		// setp2. ['a', '2', 'c] => {"a": {"2": {"c": {}}}}
		let res = "{"
		for (let j = 0; j < arr.length; j++) {
			res += '"' + arr[j] + '": {'
			var sep = "}".repeat(j + 1)
		}
		res += sep + "}"
		// step3. [{"a": {"2": {"c": {}}}}, {"a": {"b"{}}}, ...]
		array.push(JSON.parse(res))
	}

	for (let i = 0; i < array.length; i++) {
		for (let v in array[i]) {
			// 4. 解构
			answer[v] = {...array[i][v], ...answer[v]}
		}
	}

	return answer
}

"""
华为机考题1

x$y = 3*x + 2*y + 2
x#y = 2*x + 3*y + 4

输入：7#6$5#12
输出：226
解释：先计算$, 在计算#
"""

def func1(v):
    arr = [int(x) for x in v.split('$')]
    a, b = arr[0], arr[1]
    for _ in arr[2:]:
        a, b = 3 * a + b + 2, _
    return 3 * a + b + 2

def func2(l):
    l = [int(x) for x in l]
    x, y = l[0], l[1]
    for _ in l[2:]:
        x, y = 2 * x + 3 * y + 4, _
    return 2 * x + 3 * y + 4
